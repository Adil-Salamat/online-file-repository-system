const database = source("database");
const database_association = source("database_association");
const idGen = source("idGen");
/**
* Generation function that loads necessary database data, then renders the view for the user.
* @param  {} req - request variable
* @param  {} res - res variable
* @param  {} message - any message to be displayed on the page
*/
function generate(req, res, message) {
  
  database_association.association.type.selectAll().then(typeList => {
    database_association.association.info.selectAll().then(assocList => {
      res.render('association.ejs', {
        title: "Add Assocation",
        logged: req.session.user,
        access: req.session.access,
        typeList,
        assocList,
        message
      });
    });
  });
  
  
}

module.exports = {

   /**
  * 
  * Generates the page for associations. 
  * 
  * @param  {} req - request variable
  * @param  {} res - response variable
  */
  loadPage: (req, res) => {
    generate(req, res, null);
  },

  /**
  * Routing for once the user submits a form. Inputs association or association type into the database.
  * 
  * @param  {} req - request variable
  * @param  {} res - response variable
  */
  addAssociation: (req, res) => {
    
    if (req.body.assocName) {
      
      let id = idGen.new();
      let name = req.body.assocName;
      let about = req.body.assocAbout;
      let type = req.body.assoc_type;
      
      database_association.association.info.insert(id, name, about, type);
      
      generate(req, res, "Assocaiton added.");
      
    } else if (req.body.assocTypeName) {
      
      database_association.association.type.insert(idGen.new(), req.body.assocTypeName, req.body.assocTypeAbout, req.body.assocTypeRelation, req.body.assocTypeInverse);
      generate(req, res, "Assocaiton Type added.");
      
      
    } else {
      
      generate(req, res, null);
    }
    
  },
};
